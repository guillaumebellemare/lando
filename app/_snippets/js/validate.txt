<script src="<?php echo URL_ROOT; ?>scripts/jquery.validate.min.js"></script> 
<script src="<?php echo URL_ROOT; ?>scripts/jquery.validate.additional-methods.min.js"></script>
<script>
jQuery.validator.addMethod("cdnPostal", function(postal, element) {
    return this.optional(element) || 
    postal.match(/[a-zA-Z][0-9][a-zA-Z](-| |)[0-9][a-zA-Z][0-9]/);
}, "<?=$validate['zip'];?>");
</script>
<script>
$(document).ready(function() {
	// Validate signup form on keyup and submit
	$("#formID").validate({
		rules: {
			tField: "required",
			tDate: {
				required: true,
				dateISO: true
			},
			tEmail: {
				required: true,
				email: true
			},
			tPhone: {
				required: true,
				phoneUS: true
			},
			tZip: {
				required: true,
				cdnPostal: true
			},
		},
		messages: {
			tField: "<?=$validate['first_name_required'];?>",
			tDate: {
				required: "<?=$validate['date_required'];?>",
				dateISO: "<?=$validate['date'];?>"
			},
			tEmail: {
				required: "<?=$validate['email_required'];?>",
				email: "<?=$validate['email'];?>"
			},
			tPhone: {
				required: "<?=$validate['phone_required'];?>",
				phoneUS: "<?=$validate['phone'];?>"
			},
		}
	});

});
</script>
